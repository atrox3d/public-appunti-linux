drivers:
	# list drivers
	/usr/bin/lsmod

	# remove driver from memory
	rmmod drivername
		rmmod pcspkr
	OR
	modprobe -r drivername
		modprobe -r pcspkr

	# find drivers path
	ls /lib/modules/$(uname -r)

	# find driver path to load
	find /lib/modules/$(uname -r)/ -iname '*drivername*.ko*'
		find /lib/modules/$(uname -r)/ -iname '*pcspkr*.ko*'
			/lib/modules/4.19.0-17-686-pae/kernel/drivers/net/wireless/broadcom/b43/b43.ko
	# reinstall driver
	insmod driverppath
		insmod /lib/modules/4.19.0-17-686-pae/kernel/drivers/net/wireless/broadcom/b43/b43.ko
	OR
	modprobe drivername
		modprobe pcspkr

	# driver config
	ls /etc/modprobe.d
